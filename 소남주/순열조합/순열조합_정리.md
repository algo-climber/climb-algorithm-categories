import java.util.Arrays;

public class Solution {
	
	static int[] nums;
	static int N;
	
	public static void main(String[] args) {
		
		nums = new int[] {0, 1, 2};
		N = nums.length;
		
		perm(0);
		
	}
	
	static void perm(int idx) {
		if (idx == N) {
			System.out.println(Arrays.toString(nums));
			return;
		}
		
		for (int i = idx; i < N; i++) {
			swap(i, idx);
			perm(idx + 1);
			swap(i, idx);
		}
	}
	
	static void swap(int idx1, int idx2) {
		int temp = nums[idx1];
		nums[idx1] = nums[idx2];
		nums[idx2] = temp;
	}
	
}
